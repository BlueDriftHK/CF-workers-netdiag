const [client, server] = Object.values(new WebSocketPair());
server.accept();
server.addEventListener('message', event => {
    const msg = JSON.parse(event.data);
    if (msg.type === 'ping')
        server.send(JSON.stringify({ type: 'pong', t: msg.t }));
});
return new Response(null, { status: 101, webSocket: client });

快速部署

  1. 克隆仓库： git clone https://github.com/BlueDriftHK/CF-workers-netdiag.git
  2. 登录 Cloudflare： 安装 Wrangler CLI 并执行 wrangler login
  3. 创建 KV 命名空间： wrangler kv:namespace create "SPEED_HISTORY"
  4. 绑定 KV 到 Worker： 在 wrangler.toml 中配置 kv_namespaces
  5. 部署 Worker： wrangler deploy --name netsight-pro
  6. 访问： 浏览器打开您的 Worker 域名，立即体验。
详细配置请参考项目 README，支持自定义限流阈值、白名单域名、CORS 策略等。
