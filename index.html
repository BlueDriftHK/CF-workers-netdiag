一键部署到 Cloudflare Workers

# 1. 安装 Wrangler CLI
npm install -g wrangler

# 2. 登录 Cloudflare
wrangler login

# 3. 创建 wrangler.toml
name = "netsight-pro"
main = "worker.js"
compatibility_date = "2024-12-01"

# 4. 部署
wrangler deploy
